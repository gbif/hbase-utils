001package org.gbif.hbase.util;
002
003import java.util.ArrayList;
004import java.util.Arrays;
005import java.util.List;
006
007import org.apache.hadoop.hbase.KeyValue;
008import org.apache.hadoop.hbase.client.Result;
009import org.apache.hadoop.hbase.util.Bytes;
010import org.junit.Before;
011import org.junit.Test;
012
013import static org.junit.Assert.assertEquals;
014import static org.junit.Assert.assertTrue;
015
016public class ResultReaderTest {
017
018  private static final String CF1_NAME = "1";
019  private static final byte[] CF1 = Bytes.toBytes(CF1_NAME);
020  private static final String CF2_NAME = "2";
021  private static final byte[] CF2 = Bytes.toBytes(CF2_NAME);
022  private static final String INT_COL_NAME = "a";
023  private static final byte[] INT_COL = Bytes.toBytes(INT_COL_NAME);
024  private static final String DOUBLE_COL_NAME = "b";
025  private static final byte[] DOUBLE_COL = Bytes.toBytes(DOUBLE_COL_NAME);
026  private static final String LONG_COL_NAME = "c";
027  private static final byte[] LONG_COL = Bytes.toBytes(LONG_COL_NAME);
028  private static final String STRING_COL_NAME = "d";
029  private static final byte[] STRING_COL = Bytes.toBytes(STRING_COL_NAME);
030  private static final String BYTES_COL_NAME = "e";
031  private static final byte[] BYTES_COL = Bytes.toBytes(BYTES_COL_NAME);
032  private static final byte[] KEY = Bytes.toBytes("12345");
033
034  private static final int INT_VAL_1 = 1111;
035  private static final double DOUBLE_VAL_1 = 2.2222222222222222222d;
036  private static final long LONG_VAL_1 = 33333333333333l;
037  private static final String STRING_VAL_1 = "not numbers";
038  private static final byte[] BYTES_VAL_1 = Bytes.toBytes("byte values");
039
040  private static final int INT_VAL_2 = 4444;
041  private static final double DOUBLE_VAL_2 = 5.55555555555555d;
042  private static final long LONG_VAL_2 = 66666666666666666l;
043  private static final String STRING_VAL_2 = "just a string";
044  private static final byte[] BYTES_VAL_2 = Bytes.toBytes("different byte values");
045
046  private Result result = null;
047
048  @Before
049  public void setup() {
050    List<KeyValue> kvs = new ArrayList<KeyValue>();
051
052    // CF1
053    KeyValue kv = new KeyValue(KEY, CF1, INT_COL, Bytes.toBytes(INT_VAL_1));
054    kvs.add(kv);
055    kv = new KeyValue(KEY, CF1, DOUBLE_COL, Bytes.toBytes(DOUBLE_VAL_1));
056    kvs.add(kv);
057    kv = new KeyValue(KEY, CF1, LONG_COL, Bytes.toBytes(LONG_VAL_1));
058    kvs.add(kv);
059    kv = new KeyValue(KEY, CF1, STRING_COL, Bytes.toBytes(STRING_VAL_1));
060    kvs.add(kv);
061    kv = new KeyValue(KEY, CF1, BYTES_COL, BYTES_VAL_1);
062    kvs.add(kv);
063
064    // CF2
065    kv = new KeyValue(KEY, CF2, INT_COL, Bytes.toBytes(INT_VAL_2));
066    kvs.add(kv);
067    kv = new KeyValue(KEY, CF2, DOUBLE_COL, Bytes.toBytes(DOUBLE_VAL_2));
068    kvs.add(kv);
069    kv = new KeyValue(KEY, CF2, LONG_COL, Bytes.toBytes(LONG_VAL_2));
070    kvs.add(kv);
071    kv = new KeyValue(KEY, CF2, STRING_COL, Bytes.toBytes(STRING_VAL_2));
072    kvs.add(kv);
073    kv = new KeyValue(KEY, CF2, BYTES_COL, BYTES_VAL_2);
074    kvs.add(kv);
075    result = new Result(kvs);
076  }
077
078  @Test
079  public void testString() {
080
081    String test = ResultReader.getString(result, CF1_NAME, STRING_COL_NAME, null);
082    assertTrue(STRING_VAL_1.equals(test));
083
084    test = ResultReader.getString(result, CF1_NAME, "fake col", "a default value");
085    assertTrue("a default value".equals(test));
086
087    test = ResultReader.getString(result, CF2_NAME, STRING_COL_NAME, null);
088    assertTrue(STRING_VAL_2.equals(test));
089  }
090
091  @Test
092  public void testDouble() {
093    Double test = ResultReader.getDouble(result, CF1_NAME, DOUBLE_COL_NAME, null);
094    assertEquals(DOUBLE_VAL_1, test, 0.00001);
095
096    test = ResultReader.getDouble(result, CF1_NAME, "fake col", 123456.789d);
097    assertEquals(Double.valueOf(123456.789), test, 0.00001);
098
099    test = ResultReader.getDouble(result, CF2_NAME, DOUBLE_COL_NAME, null);
100    assertEquals(DOUBLE_VAL_2, test, 0.00001);
101  }
102
103  @Test
104  public void testInteger() {
105    Integer test = ResultReader.getInteger(result, CF1_NAME, INT_COL_NAME, null);
106    assertEquals(INT_VAL_1, test.intValue());
107
108    test = ResultReader.getInteger(result, CF1_NAME, "fake col", 123456);
109    assertEquals(Integer.valueOf(123456), test);
110
111    test = ResultReader.getInteger(result, CF2_NAME, INT_COL_NAME, null);
112    assertEquals(INT_VAL_2, test.intValue());
113  }
114
115  @Test
116  public void testLong() {
117    Long test = ResultReader.getLong(result, CF1_NAME, LONG_COL_NAME, null);
118    assertEquals(LONG_VAL_1, test.longValue());
119
120    test = ResultReader.getLong(result, CF1_NAME, "fake col", 1234567890l);
121    assertEquals(Long.valueOf(1234567890l), test);
122
123    test = ResultReader.getLong(result, CF2_NAME, LONG_COL_NAME, null);
124    assertEquals(LONG_VAL_2, test.longValue());
125  }
126
127  @Test
128  public void testBytes() {
129    byte[] test = ResultReader.getBytes(result, CF1_NAME, BYTES_COL_NAME, null);
130    assertTrue(Arrays.equals(BYTES_VAL_1, test));
131
132    test = ResultReader.getBytes(result, CF1_NAME, "fake col", Bytes.toBytes(1234567890l));
133    assertTrue(Arrays.equals(Bytes.toBytes(1234567890l), test));
134
135    test = ResultReader.getBytes(result, CF2_NAME, BYTES_COL_NAME, null);
136    assertTrue(Arrays.equals(BYTES_VAL_2, test));
137  }
138}
