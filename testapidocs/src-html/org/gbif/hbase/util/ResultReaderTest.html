014package org.gbif.hbase.util;
015
016import java.util.ArrayList;
017import java.util.Arrays;
018import java.util.List;
019
020import org.apache.hadoop.hbase.Cell;
021import org.apache.hadoop.hbase.CellBuilderFactory;
022import org.apache.hadoop.hbase.CellBuilderType;
023import org.apache.hadoop.hbase.client.Result;
024import org.apache.hadoop.hbase.util.Bytes;
025import org.junit.Before;
026import org.junit.Test;
027
028import static org.junit.Assert.assertEquals;
029import static org.junit.Assert.assertTrue;
030
031public class ResultReaderTest {
032
033  private static final String CF1_NAME = "1";
034  private static final byte[] CF1 = Bytes.toBytes(CF1_NAME);
035  private static final String CF2_NAME = "2";
036  private static final byte[] CF2 = Bytes.toBytes(CF2_NAME);
037  private static final String INT_COL_NAME = "a";
038  private static final byte[] INT_COL = Bytes.toBytes(INT_COL_NAME);
039  private static final String DOUBLE_COL_NAME = "b";
040  private static final byte[] DOUBLE_COL = Bytes.toBytes(DOUBLE_COL_NAME);
041  private static final String LONG_COL_NAME = "c";
042  private static final byte[] LONG_COL = Bytes.toBytes(LONG_COL_NAME);
043  private static final String STRING_COL_NAME = "d";
044  private static final byte[] STRING_COL = Bytes.toBytes(STRING_COL_NAME);
045  private static final String BYTES_COL_NAME = "e";
046  private static final byte[] BYTES_COL = Bytes.toBytes(BYTES_COL_NAME);
047  private static final byte[] KEY = Bytes.toBytes("12345");
048
049  private static final int INT_VAL_1 = 1111;
050  private static final double DOUBLE_VAL_1 = 2.2222222222222222222d;
051  private static final long LONG_VAL_1 = 33333333333333l;
052  private static final String STRING_VAL_1 = "not numbers";
053  private static final byte[] BYTES_VAL_1 = Bytes.toBytes("byte values");
054
055  private static final int INT_VAL_2 = 4444;
056  private static final double DOUBLE_VAL_2 = 5.55555555555555d;
057  private static final long LONG_VAL_2 = 66666666666666666l;
058  private static final String STRING_VAL_2 = "just a string";
059  private static final byte[] BYTES_VAL_2 = Bytes.toBytes("different byte values");
060
061  private Result result = null;
062
063  private static Cell newCell(byte[] row, byte[] columnFamily, byte[] qualifier, byte[] value) {
064    return CellBuilderFactory.create(CellBuilderType.SHALLOW_COPY)
065            .setRow(row)
066            .setFamily(columnFamily)
067            .setQualifier(qualifier)
068            .setValue(value)
069            .setType(Cell.Type.Put)
070            .build();
071  }
072  @Before
073  public void setup() {
074    List<Cell> cells = new ArrayList<>();
075
076    // CF1
077    Cell cell = newCell(KEY, CF1, INT_COL, Bytes.toBytes(INT_VAL_1));
078    cells.add(cell);
079    cell = newCell(KEY, CF1, DOUBLE_COL, Bytes.toBytes(DOUBLE_VAL_1));
080    cells.add(cell);
081    cell = newCell(KEY, CF1, LONG_COL, Bytes.toBytes(LONG_VAL_1));
082    cells.add(cell);
083    cell = newCell(KEY, CF1, STRING_COL, Bytes.toBytes(STRING_VAL_1));
084    cells.add(cell);
085    cell = newCell(KEY, CF1, BYTES_COL, BYTES_VAL_1);
086    cells.add(cell);
087
088    // CF2
089    cell = newCell(KEY, CF2, INT_COL, Bytes.toBytes(INT_VAL_2));
090    cells.add(cell);
091    cell = newCell(KEY, CF2, DOUBLE_COL, Bytes.toBytes(DOUBLE_VAL_2));
092    cells.add(cell);
093    cell = newCell(KEY, CF2, LONG_COL, Bytes.toBytes(LONG_VAL_2));
094    cells.add(cell);
095    cell = newCell(KEY, CF2, STRING_COL, Bytes.toBytes(STRING_VAL_2));
096    cells.add(cell);
097    cell = newCell(KEY, CF2, BYTES_COL, BYTES_VAL_2);
098    cells.add(cell);
099    result = Result.create(cells);
100  }
101
102  @Test
103  public void testString() {
104
105    String test = ResultReader.getString(result, CF1_NAME, STRING_COL_NAME, null);
106    assertTrue(STRING_VAL_1.equals(test));
107
108    test = ResultReader.getString(result, CF1_NAME, "fake col", "a default value");
109    assertTrue("a default value".equals(test));
110
111    test = ResultReader.getString(result, CF2_NAME, STRING_COL_NAME, null);
112    assertTrue(STRING_VAL_2.equals(test));
113  }
114
115  @Test
116  public void testDouble() {
117    Double test = ResultReader.getDouble(result, CF1_NAME, DOUBLE_COL_NAME, null);
118    assertEquals(DOUBLE_VAL_1, test, 0.00001);
119
120    test = ResultReader.getDouble(result, CF1_NAME, "fake col", 123456.789d);
121    assertEquals(123456.789, test, 0.00001);
122
123    test = ResultReader.getDouble(result, CF2_NAME, DOUBLE_COL_NAME, null);
124    assertEquals(DOUBLE_VAL_2, test, 0.00001);
125  }
126
127  @Test
128  public void testInteger() {
129    Integer test = ResultReader.getInteger(result, CF1_NAME, INT_COL_NAME, null);
130    assertEquals(INT_VAL_1, test.intValue());
131
132    test = ResultReader.getInteger(result, CF1_NAME, "fake col", 123456);
133    assertEquals(Integer.valueOf(123456), test);
134
135    test = ResultReader.getInteger(result, CF2_NAME, INT_COL_NAME, null);
136    assertEquals(INT_VAL_2, test.intValue());
137  }
138
139  @Test
140  public void testLong() {
141    Long test = ResultReader.getLong(result, CF1_NAME, LONG_COL_NAME, null);
142    assertEquals(LONG_VAL_1, test.longValue());
143
144    test = ResultReader.getLong(result, CF1_NAME, "fake col", 1234567890l);
145    assertEquals(Long.valueOf(1234567890l), test);
146
147    test = ResultReader.getLong(result, CF2_NAME, LONG_COL_NAME, null);
148    assertEquals(LONG_VAL_2, test.longValue());
149  }
150
151  @Test
152  public void testBytes() {
153    byte[] test = ResultReader.getBytes(result, CF1_NAME, BYTES_COL_NAME, null);
154    assertTrue(Arrays.equals(BYTES_VAL_1, test));
155
156    test = ResultReader.getBytes(result, CF1_NAME, "fake col", Bytes.toBytes(1234567890l));
157    assertTrue(Arrays.equals(Bytes.toBytes(1234567890l), test));
158
159    test = ResultReader.getBytes(result, CF2_NAME, BYTES_COL_NAME, null);
160    assertTrue(Arrays.equals(BYTES_VAL_2, test));
161  }
162}
