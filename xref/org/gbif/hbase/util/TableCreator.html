32    /**
33     * Create an hbase table with given colfamily, using the given splits file to presplit the table.  Regions are set
34     * to the given max regionSize (given in MB and translated into bytes). Input table
35     *
36     * @param tableName      table to create
37     * @param colFamily      col family to create
38     * @param splitsFileName one split per line, interpreted as ints
39     * @param regionSize     max region file size in MB
40     */
41    private void createPreSplitTable(String tableName, String colFamily, String splitsFileName, Integer regionSize) {
42      byte[][] byteSplits = null;
43  
44      if (splitsFileName != null) {
45        // assumes splits are written as strings but are to be interpreted as ints
46        log.info("Reading splits file [{}]", splitsFileName);
47        List<Integer> splits = new ArrayList<Integer>();
48        try {
49          BufferedReader reader =
50            new BufferedReader(new InputStreamReader(new FileInputStream(new File(splitsFileName)), "UTF-8"));
51          while (reader.ready()) {
52            String line = reader.readLine();
53            splits.add(Integer.valueOf(line));
54          }
55        } catch (FileNotFoundException e) {
56          log.warn("Splits file not found - exiting", e);
57          System.exit(1);
58        } catch (UnsupportedEncodingException e) {
59          log.error("Drastic jvm failure - exiting", e);
60          System.exit(1);
61        } catch (IOException e) {
62          log.warn("Failure reading splits file - exiting", e);
63          System.exit(1);
64        }
65        log.info("Loaded [{}] splits", splits.size());
66  
67        byteSplits = new byte[splits.size()][];
68        for (int i = 0; i < splits.size(); i++) {
69          byteSplits[i] = Bytes.toBytes(splits.get(i));
70        }
71      }
72  
73      log.info("Creating table");
74      Configuration config = HBaseConfiguration.create();
75      try {
76        HBaseAdmin admin = new HBaseAdmin(config);
77        HColumnDescriptor cf = new HColumnDescriptor(colFamily);
78        cf.setCompressionType(Compression.Algorithm.SNAPPY);
79        cf.setDataBlockEncoding(DataBlockEncoding.FAST_DIFF);
80        cf.setMaxVersions(1);
81        HTableDescriptor ror = new HTableDescriptor(tableName);
82        if (regionSize != null && regionSize > 0) ror.setMaxFileSize(regionSize * 1048576); // 1 MB = 2^20 bytes
83        ror.addFamily(cf);
84        if (byteSplits == null) {
85          admin.createTable(ror);
86        }
87        else {
88          admin.createTable(ror, byteSplits);
89        }
90        admin.close();
91      } catch (MasterNotRunningException e) {
92        log.info("HBase exception", e);
93        System.exit(1);
94      } catch (ZooKeeperConnectionException e) {
95        log.info("HBase exception", e);
96        System.exit(1);
97      } catch (IOException e) {
98        log.info("HBase exception", e);
99        System.exit(1);
100     }
101     log.info("Table created - exiting");
102   }

104   public static void main(String[] args) {
105     if (args.length < 2) {
106       System.out.println("Usage: TableCreator <table_name> <column_family> [<splits_file>] [<region_file_size_mb>]");
107       System.exit(1);
108     }
109     String tableName = args[0];
110     String colFam = args[1];
111     String splitsFile = args.length > 2 ? args[2] : null;
112     Integer regionSize = args.length == 4 ? Integer.valueOf(args[3]) : null;
113 
114     TableCreator instance = new TableCreator();
115     //    instance.createPreSplitTable("ror_120", "v", "/Users/oliver/SourceCode/ror_120_splits.txt", 256*3);
116     instance.createPreSplitTable(tableName, colFam, splitsFile, regionSize);
117   }
