7   import org.apache.hadoop.conf.Configuration;
8   import org.apache.hadoop.hbase.HBaseConfiguration;
9   import org.apache.hadoop.hbase.client.HTable;
10  import org.apache.hadoop.hbase.util.Bytes;
11  import org.slf4j.Logger;
12  import org.slf4j.LoggerFactory;
13  
14  public class TableSplitReader {
15  
16    private static final Logger LOG = LoggerFactory.getLogger(TableCloneCreator.class);
17  
18    // not to be instantiated
19    private TableSplitReader() {
20    }
21  
22    public static void main(String[] args) throws IOException {
23      if (args.length != 1) {
24        LOG.error("Usage: org.gbif.hbase.util.TableSplitReader <existing table>");
25        System.exit(1);
26      }
27  
28      readIntSplits(args[0]);
29    }
30  
31    private static void readIntSplits(String existingTableName) throws IOException {
32      Configuration config = HBaseConfiguration.create();
33      HTable existingTable = new HTable(config, Bytes.toBytes(existingTableName));
34  
35      String fileName = existingTableName + "_splits.txt";
36      LOG.info("Writing file [{}]", fileName);
37      File outFile = new File(fileName);
38      FileWriter fileWriter = new FileWriter(outFile);
39      byte[][] existingStartKeys = existingTable.getStartKeys();
40      for (int i = 1; i < existingStartKeys.length; i++) {
41        fileWriter.write(Bytes.toInt(existingStartKeys[i]) + "\n");
42      }
43  
44      fileWriter.close();
45    }
46  }
