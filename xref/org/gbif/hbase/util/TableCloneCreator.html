4   import java.util.Arrays;
5   import java.util.Collection;

7   import org.apache.hadoop.conf.Configuration;
8   import org.apache.hadoop.hbase.HBaseConfiguration;
9   import org.apache.hadoop.hbase.HColumnDescriptor;
10  import org.apache.hadoop.hbase.HTableDescriptor;
11  import org.apache.hadoop.hbase.client.HBaseAdmin;
12  import org.apache.hadoop.hbase.client.HTable;
13  import org.apache.hadoop.hbase.util.Bytes;
14  import org.slf4j.Logger;
15  import org.slf4j.LoggerFactory;
16  
17  /**
18   * Creates a new table in HBase using the splits from the existing table. Specify -Pprod or -Pdev when building to
19   * point to the right cluster!
20   */
21  public class TableCloneCreator {
22  
23    private static final Logger LOG = LoggerFactory.getLogger(TableCloneCreator.class);
24  
25    // not to be instantiated
26    private TableCloneCreator() {}
27  
28    public static void main(String[] args) throws IOException {
29      if (args.length != 2) {
30        LOG.error("Usage: org.gbif.hbase.util.TableCloneCreator <existing table> <new table to create>");
31        System.exit(1);
32      }
33  
34      createPresplitFromExisting(args[0], args[1]);
35    }
36  
37    private static void createPresplitFromExisting(String existingTableName, String newTable) throws IOException {
38      Configuration config = HBaseConfiguration.create();
39      HTable existingTable = new HTable(config, Bytes.toBytes(existingTableName));
40      Collection<HColumnDescriptor> existingColFams = existingTable.getTableDescriptor().getFamilies();
41  
42      HBaseAdmin admin = new HBaseAdmin(config);
43      HTableDescriptor tableDescriptor = new HTableDescriptor(newTable);
44      tableDescriptor.setMaxFileSize(existingTable.getTableDescriptor().getMaxFileSize());
45      for (HColumnDescriptor colFam : existingColFams) {
46        tableDescriptor.addFamily(colFam);
47      }
48      byte[][] existingStartKeys = existingTable.getStartKeys();
49      // first key in getStartKeys will be empty
50      byte[][] splits = Arrays.copyOfRange(existingStartKeys, 1, existingStartKeys.length);
51  
52      admin.createTable(tableDescriptor, splits);
53      admin.close();
54    }
55  }
