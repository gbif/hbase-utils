001package org.gbif.hbase.util;
002
003import java.io.IOException;
004import java.util.Arrays;
005import java.util.Collection;
006
007import org.apache.hadoop.conf.Configuration;
008import org.apache.hadoop.hbase.HBaseConfiguration;
009import org.apache.hadoop.hbase.HColumnDescriptor;
010import org.apache.hadoop.hbase.HTableDescriptor;
011import org.apache.hadoop.hbase.client.HBaseAdmin;
012import org.apache.hadoop.hbase.client.HTable;
013import org.apache.hadoop.hbase.util.Bytes;
014import org.slf4j.Logger;
015import org.slf4j.LoggerFactory;
016
017/**
018 * Creates a new table in HBase using the splits from the existing table. Specify -Pprod or -Pdev when building to
019 * point to the right cluster!
020 */
021public class TableCloneCreator {
022
023  private static final Logger LOG = LoggerFactory.getLogger(TableCloneCreator.class);
024
025  // not to be instantiated
026  private TableCloneCreator() {}
027
028  public static void main(String[] args) throws IOException {
029    if (args.length != 2) {
030      LOG.error("Usage: org.gbif.hbase.util.TableCloneCreator <existing table> <new table to create>");
031      System.exit(1);
032    }
033
034    createPresplitFromExisting(args[0], args[1]);
035  }
036
037  private static void createPresplitFromExisting(String existingTableName, String newTable) throws IOException {
038    Configuration config = HBaseConfiguration.create();
039    HTable existingTable = new HTable(config, Bytes.toBytes(existingTableName));
040    Collection<HColumnDescriptor> existingColFams = existingTable.getTableDescriptor().getFamilies();
041
042    HBaseAdmin admin = new HBaseAdmin(config);
043    HTableDescriptor tableDescriptor = new HTableDescriptor(newTable);
044    tableDescriptor.setMaxFileSize(existingTable.getTableDescriptor().getMaxFileSize());
045    for (HColumnDescriptor colFam : existingColFams) {
046      tableDescriptor.addFamily(colFam);
047    }
048    byte[][] existingStartKeys = existingTable.getStartKeys();
049    // first key in getStartKeys will be empty
050    byte[][] splits = Arrays.copyOfRange(existingStartKeys, 1, existingStartKeys.length);
051
052    admin.createTable(tableDescriptor, splits);
053    admin.close();
054  }
055}
