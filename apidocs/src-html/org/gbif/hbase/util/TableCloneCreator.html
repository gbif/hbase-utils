014package org.gbif.hbase.util;
015
016import java.io.IOException;
017import java.util.Arrays;
018
019import org.apache.hadoop.hbase.HBaseConfiguration;
020import org.apache.hadoop.hbase.TableName;
021import org.apache.hadoop.hbase.client.Admin;
022import org.apache.hadoop.hbase.client.Connection;
023import org.apache.hadoop.hbase.client.ConnectionFactory;
024import org.apache.hadoop.hbase.client.Table;
025import org.apache.hadoop.hbase.client.TableDescriptorBuilder;
026import org.slf4j.Logger;
027import org.slf4j.LoggerFactory;
028
029/**
030 * Creates a new table in HBase using the splits from the existing table. Specify -Pprod or -Pdev when building to
031 * point to the right cluster!
032 */
033public class TableCloneCreator {
034
035  private static final Logger LOG = LoggerFactory.getLogger(TableCloneCreator.class);
036
037  // not to be instantiated
038  private TableCloneCreator() {}
039
040  public static void main(String[] args) throws IOException {
041    if (args.length != 2) {
042      LOG.error("Usage: org.gbif.hbase.util.TableCloneCreator <existing table> <new table to create>");
043      System.exit(1);
044    }
045
046    createPreSplitFromExisting(args[0], args[1]);
047  }
048
049  private static void createPreSplitFromExisting(String existingTableName, String newTable) throws IOException {
050    try (Connection connection = ConnectionFactory.createConnection(HBaseConfiguration.create());
051         Table existingTable = connection.getTable(TableName.valueOf(existingTableName));
052         Admin admin = connection.getAdmin()) {
053
054      TableDescriptorBuilder tableDescriptor = TableDescriptorBuilder.newBuilder(TableName.valueOf(newTable))
055                                                .setMaxFileSize(existingTable.getDescriptor().getMaxFileSize());
056      tableDescriptor.setColumnFamilies(Arrays.asList(existingTable.getDescriptor().getColumnFamilies()));
057
058      byte[][] existingStartKeys = existingTable.getRegionLocator().getStartKeys();
059      // first key in getStartKeys will be empty
060      byte[][] splits = Arrays.copyOfRange(existingStartKeys, 1, existingStartKeys.length);
061
062      admin.createTable(tableDescriptor.build(), splits);
063    }
064  }
065}
