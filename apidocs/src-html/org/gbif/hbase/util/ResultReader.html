014package org.gbif.hbase.util;
015
016import org.apache.hadoop.hbase.Cell;
017import org.apache.hadoop.hbase.client.Result;
018import org.apache.hadoop.hbase.util.Bytes;
019
020/**
021 * Static utility methods for reading (Java) typed fields from an HBase Result.
022 */
023public class ResultReader {
024
025  /**
026   * Should never be constructed.
027   */
028  private ResultReader() {
029  }
030
031  /**
032   * Read the value of this cell and interpret as String.
033   *
034   * @param row          the HBase Result from which to read
035   * @param columnFamily column family that holds the column
036   * @param columnName   column or "qualifier"
037   * @param defaultValue returned if value at columnName is null
038   *
039   * @return the value from the specified column, or defaultValue if it's null/doesn't exist
040   */
041  public static String getString(Result row, String columnFamily, String columnName, String defaultValue) {
042    byte[] raw = row.getValue(Bytes.toBytes(columnFamily), Bytes.toBytes(columnName));
043    return  (raw == null) ? defaultValue : Bytes.toString(raw);
044  }
045
046  /**
047   * Read the value of this cell and interpret as Integer.
048   *
049   * @param row          the HBase Result from which to read
050   * @param columnFamily column family that holds the column
051   * @param columnName   column or "qualifier"
052   * @param defaultValue returned if value at columnName is null
053   *
054   * @return the value from the specified column, or defaultValue if it's null/doesn't exist
055   */
056  public static Integer getInteger(Result row, String columnFamily, String columnName, Integer defaultValue) {
057    byte[] raw = row.getValue(Bytes.toBytes(columnFamily), Bytes.toBytes(columnName));
058    return (raw == null) ? defaultValue : Integer.valueOf(Bytes.toInt(raw));
059  }
060
061  /**
062   * Read the value of this cell and interpret as Long.
063   *
064   * @param row          the HBase Result from which to read
065   * @param columnFamily column family that holds the column
066   * @param columnName   column or "qualifier"
067   * @param defaultValue returned if value at columnName is null
068   *
069   * @return the value from the specified column, or defaultValue if it's null/doesn't exist
070   */
071  public static Long getLong(Result row, String columnFamily, String columnName, Long defaultValue) {
072    byte[] raw = row.getValue(Bytes.toBytes(columnFamily), Bytes.toBytes(columnName));
073    return (raw == null) ? defaultValue : Long.valueOf(Bytes.toLong(raw));
074  }
075
076  /**
077   * Read the value of this cell and interpret as Float.
078   *
079   * @param row          the HBase Result from which to read
080   * @param columnFamily column family that holds the column
081   * @param columnName   column or "qualifier"
082   * @param defaultValue returned if value at columnName is null
083   *
084   * @return the value from the specified column, or defaultValue if it's null/doesn't exist
085   */
086  public static Float getFloat(Result row, String columnFamily, String columnName, Float defaultValue) {
087    byte[] raw = row.getValue(Bytes.toBytes(columnFamily), Bytes.toBytes(columnName));
088    return (raw == null) ? defaultValue : Float.valueOf(Bytes.toFloat(raw));
089  }
090
091  /**
092   * Read the value of this cell and interpret as Double.
093   *
094   * @param row          the HBase Result from which to read
095   * @param columnFamily column family that holds the column
096   * @param columnName   column or "qualifier"
097   * @param defaultValue returned if value at columnName is null
098   *
099   * @return the value from the specified column, or defaultValue if it's null/doesn't exist
100   */
101  public static Double getDouble(Result row, String columnFamily, String columnName, Double defaultValue) {
102    byte[] raw = row.getValue(Bytes.toBytes(columnFamily), Bytes.toBytes(columnName));
103    return (raw == null) ? defaultValue : Double.valueOf(Bytes.toDouble(raw));
104  }
105
106  /**
107   * Read the value of this cell and return it uninterpreted as byte[].
108   *
109   * @param row          the HBase Result from which to read
110   * @param columnFamily column family that holds the column
111   * @param columnName   column or "qualifier"
112   * @param defaultValue returned if value at columnName is null
113   *
114   * @return the value from the specified column, or defaultValue if it's null/doesn't exist
115   */
116  public static byte[] getBytes(Result row, String columnFamily, String columnName, byte[] defaultValue) {
117    byte[] raw = row.getValue(Bytes.toBytes(columnFamily), Bytes.toBytes(columnName));
118    return (raw == null) ? defaultValue : raw;
119  }
120
121  // TODO: @nullable
122  public static Long getTimestamp(Result row, String columnFamily, String columnName) {
123    Cell raw = row.getColumnLatestCell(Bytes.toBytes(columnFamily), Bytes.toBytes(columnName));
124    return (raw == null) ? null : raw.getTimestamp();
125  }
126}
