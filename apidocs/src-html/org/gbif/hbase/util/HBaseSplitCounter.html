001package org.gbif.hbase.util;
002
003import java.io.File;
004import java.io.FileNotFoundException;
005import java.io.FileOutputStream;
006import java.io.IOException;
007import java.io.OutputStreamWriter;
008import java.util.NavigableMap;
009
010import org.apache.hadoop.conf.Configuration;
011import org.apache.hadoop.hbase.HBaseConfiguration;
012import org.apache.hadoop.hbase.HRegionInfo;
013import org.apache.hadoop.hbase.ServerName;
014import org.apache.hadoop.hbase.TableName;
015import org.apache.hadoop.hbase.client.MetaScanner;
016import org.apache.hadoop.hbase.util.Bytes;
017import org.slf4j.Logger;
018import org.slf4j.LoggerFactory;
019
020public class HBaseSplitCounter {
021
022  private static final Logger LOG = LoggerFactory.getLogger(HBaseSplitCounter.class);
023
024  private static void generateSplitsFile(String tableName, String outFileName) throws IOException {
025    LOG.info("Finding splits for [{}]", tableName);
026
027    FileOutputStream fos = null;
028    OutputStreamWriter writer = null;
029    try {
030      File outFile = new File(outFileName);
031      outFile.delete();
032      outFile.createNewFile();
033      fos = new FileOutputStream(outFile);
034      writer = new OutputStreamWriter(fos, "UTF-8");
035
036      Configuration config = HBaseConfiguration.create();
037      TableName tableNameBytes = TableName.valueOf(tableName);
038      NavigableMap<HRegionInfo, ServerName> regions =
039        MetaScanner.allTableRegions(config, null, tableNameBytes, false);
040      int goodSplits = 0;
041      for (HRegionInfo region : regions.keySet()) {
042        byte[] endKey = region.getEndKey();
043        if (endKey.length > 0) {
044          int goodKey = Bytes.toInt(endKey);
045          goodSplits++;
046          writer.write(goodKey + "\n");
047          writer.flush();
048          LOG.info("Got endkey [{}]", goodKey);
049        }
050      }
051
052      LOG.info("Finished getting [{}] endkey splits for [{}] regions in [{}]", goodSplits, regions.size(), tableName);
053    } catch (FileNotFoundException e) {
054      LOG.error("Couldn't create file [{}] - aborting", outFileName, e);
055      System.exit(1);
056    } catch (IOException e) {
057      LOG.warn("IOException when writing to file", e);
058    } finally {
059      try {
060        if (writer != null) {
061          writer.close();
062        }
063        if (fos != null) {
064          fos.close();
065        }
066      } catch (IOException e) {
067        LOG.warn("Couldn't close file connections", e);
068      }
069    }
070  }
071
072  public static void main(String[] args) throws IOException {
073    if (args.length != 2) {
074      LOG.error("Usage: HBaseSplitCounter <tableName> <outputFileName>");
075      System.exit(1);
076    }
077    generateSplitsFile(args[0], args[1]);
078  }
079}
