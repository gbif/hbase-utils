014package org.gbif.hbase.util;
015
016import java.io.File;
017import java.io.FileNotFoundException;
018import java.io.FileOutputStream;
019import java.io.IOException;
020import java.io.OutputStreamWriter;
021import java.nio.charset.StandardCharsets;
022import java.nio.file.Files;
023import java.util.List;
024
025import org.apache.hadoop.conf.Configuration;
026import org.apache.hadoop.hbase.HBaseConfiguration;
027import org.apache.hadoop.hbase.HRegionLocation;
028import org.apache.hadoop.hbase.TableName;
029import org.apache.hadoop.hbase.client.Connection;
030import org.apache.hadoop.hbase.client.ConnectionFactory;
031import org.apache.hadoop.hbase.client.RegionLocator;
032import org.apache.hadoop.hbase.util.Bytes;
033import org.slf4j.Logger;
034import org.slf4j.LoggerFactory;
035
036public class HBaseSplitCounter {
037
038  private static final Logger LOG = LoggerFactory.getLogger(HBaseSplitCounter.class);
039
040  private static File reCreate(String fileName) throws IOException {
041    File file = new File(fileName);
042    Files.deleteIfExists(file.toPath());
043    Files.createFile(file.toPath());
044    return file;
045  }
046
047  private static void generateSplitsFile(String tableName, String outFileName) throws IOException {
048    LOG.info("Finding splits for [{}]", tableName);
049
050    File outFile = reCreate(outFileName);
051
052    try (FileOutputStream fos = new FileOutputStream(outFile);
053         OutputStreamWriter writer = new OutputStreamWriter(fos, StandardCharsets.UTF_8)) {
054      Configuration config = HBaseConfiguration.create();
055      TableName hTableName = TableName.valueOf(tableName);
056      try(Connection connection = ConnectionFactory.createConnection(config);
057          RegionLocator regionLocator = connection.getRegionLocator(hTableName)) {
058        List<HRegionLocation> regions = regionLocator.getAllRegionLocations();
059        int goodSplits = 0;
060        for (HRegionLocation region : regions) {
061          byte[] endKey = region.getRegion().getEndKey();
062          if (endKey.length > 0) {
063            int goodKey = Bytes.toInt(endKey);
064            goodSplits++;
065            writer.write(goodKey + "\n");
066            writer.flush();
067            LOG.info("Got endkey [{}]", goodKey);
068          }
069        }
070        LOG.info("Finished getting [{}] endkey splits for [{}] regions in [{}]", goodSplits, regions.size(), tableName);
071      }
072    } catch (FileNotFoundException e) {
073      LOG.error("Couldn't create file [{}] - aborting", outFileName, e);
074      System.exit(1);
075    } catch (IOException e) {
076      LOG.warn("IOException when writing to file", e);
077    }
078  }
079
080  public static void main(String[] args) throws IOException {
081    if (args.length != 2) {
082      LOG.error("Usage: HBaseSplitCounter <tableName> <outputFileName>");
083      System.exit(1);
084    }
085    generateSplitsFile(args[0], args[1]);
086  }
087}
