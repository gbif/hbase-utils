001package org.gbif.hbase.util;
002
003import java.io.File;
004import java.io.FileWriter;
005import java.io.IOException;
006
007import org.apache.hadoop.conf.Configuration;
008import org.apache.hadoop.hbase.HBaseConfiguration;
009import org.apache.hadoop.hbase.client.HTable;
010import org.apache.hadoop.hbase.util.Bytes;
011import org.slf4j.Logger;
012import org.slf4j.LoggerFactory;
013
014public class TableSplitReader {
015
016  private static final Logger LOG = LoggerFactory.getLogger(TableCloneCreator.class);
017
018  // not to be instantiated
019  private TableSplitReader() {
020  }
021
022  public static void main(String[] args) throws IOException {
023    if (args.length != 1) {
024      LOG.error("Usage: org.gbif.hbase.util.TableSplitReader <existing table>");
025      System.exit(1);
026    }
027
028    readIntSplits(args[0]);
029  }
030
031  private static void readIntSplits(String existingTableName) throws IOException {
032    Configuration config = HBaseConfiguration.create();
033    HTable existingTable = new HTable(config, Bytes.toBytes(existingTableName));
034
035    String fileName = existingTableName + "_splits.txt";
036    LOG.info("Writing file [{}]", fileName);
037    File outFile = new File(fileName);
038    FileWriter fileWriter = new FileWriter(outFile);
039    byte[][] existingStartKeys = existingTable.getStartKeys();
040    for (int i = 1; i < existingStartKeys.length; i++) {
041      fileWriter.write(Bytes.toInt(existingStartKeys[i]) + "\n");
042    }
043
044    fileWriter.close();
045  }
046}
