014package org.gbif.hbase.util;
015
016import java.io.File;
017import java.io.FileWriter;
018import java.io.IOException;
019
020import org.apache.hadoop.conf.Configuration;
021import org.apache.hadoop.hbase.HBaseConfiguration;
022import org.apache.hadoop.hbase.TableName;
023import org.apache.hadoop.hbase.client.Connection;
024import org.apache.hadoop.hbase.client.ConnectionFactory;
025import org.apache.hadoop.hbase.client.Table;
026import org.apache.hadoop.hbase.util.Bytes;
027import org.slf4j.Logger;
028import org.slf4j.LoggerFactory;
029
030public class TableSplitReader {
031
032  private static final Logger LOG = LoggerFactory.getLogger(TableCloneCreator.class);
033
034  // not to be instantiated
035  private TableSplitReader() {
036  }
037
038  public static void main(String[] args) throws IOException {
039    if (args.length != 1) {
040      LOG.error("Usage: org.gbif.hbase.util.TableSplitReader <existing table>");
041      System.exit(1);
042    }
043
044    readIntSplits(args[0]);
045  }
046
047  private static File splitsFile(String tableName) {
048    String fileName = tableName + "_splits.txt";
049    return new File(fileName);
050  }
051  private static void readIntSplits(String existingTableName) throws IOException {
052    Configuration config = HBaseConfiguration.create();
053    File outFile = splitsFile(existingTableName);
054    try (FileWriter fileWriter = new FileWriter(outFile);
055         Connection connection = ConnectionFactory.createConnection(config);
056         Table existingTable = connection.getTable(TableName.valueOf(existingTableName))) {
057      LOG.info("Writing file [{}]", outFile.getName());
058      byte[][] existingStartKeys = existingTable.getRegionLocator().getStartKeys();
059      for (int i = 1; i < existingStartKeys.length; i++) {
060        fileWriter.write(Bytes.toInt(existingStartKeys[i]) + "\n");
061      }
062  }
063  }
064}
