014package org.gbif.hbase.util;
015
016import java.io.BufferedReader;
017import java.io.File;
018import java.io.FileInputStream;
019import java.io.FileNotFoundException;
020import java.io.IOException;
021import java.io.InputStreamReader;
022import java.io.UnsupportedEncodingException;
023import java.nio.charset.StandardCharsets;
024import java.util.ArrayList;
025import java.util.List;
026
027import org.apache.hadoop.hbase.HBaseConfiguration;
028import org.apache.hadoop.hbase.TableName;
029import org.apache.hadoop.hbase.client.Admin;
030import org.apache.hadoop.hbase.client.ColumnFamilyDescriptor;
031import org.apache.hadoop.hbase.client.ColumnFamilyDescriptorBuilder;
032import org.apache.hadoop.hbase.client.Connection;
033import org.apache.hadoop.hbase.client.ConnectionFactory;
034import org.apache.hadoop.hbase.client.TableDescriptorBuilder;
035import org.apache.hadoop.hbase.io.compress.Compression;
036import org.apache.hadoop.hbase.io.encoding.DataBlockEncoding;
037import org.apache.hadoop.hbase.util.Bytes;
038import org.slf4j.Logger;
039import org.slf4j.LoggerFactory;
040
041public class TableCreator {
042
043  private static Logger log = LoggerFactory.getLogger(org.gbif.hbase.util.TableCreator.class);
044
045  /**
046   * Create an hbase table with given colfamily, using the given splits file to presplit the table.  Regions are set
047   * to the given max regionSize (given in MB and translated into bytes). Input table
048   *
049   * @param tableName      table to create
050   * @param colFamily      col family to create
051   * @param splitsFileName one split per line, interpreted as ints
052   * @param regionSize     max region file size in MB
053   */
054  private void createPreSplitTable(String tableName, String colFamily, String splitsFileName, Integer regionSize) {
055    byte[][] byteSplits = null;
056
057    if (splitsFileName != null) {
058      // assumes splits are written as strings but are to be interpreted as ints
059      log.info("Reading splits file [{}]", splitsFileName);
060      List<Integer> splits = new ArrayList<Integer>();
061      try {
062        BufferedReader reader =
063          new BufferedReader(new InputStreamReader(new FileInputStream(new File(splitsFileName)), "UTF-8"));
064        while (reader.ready()) {
065          String line = reader.readLine();
066          splits.add(Integer.valueOf(line));
067        }
068      } catch (FileNotFoundException e) {
069        log.warn("Splits file not found - exiting", e);
070        System.exit(1);
071      } catch (UnsupportedEncodingException e) {
072        log.error("Drastic jvm failure - exiting", e);
073        System.exit(1);
074      } catch (IOException e) {
075        log.warn("Failure reading splits file - exiting", e);
076        System.exit(1);
077      }
078      log.info("Loaded [{}] splits", splits.size());
079
080      byteSplits = new byte[splits.size()][];
081      for (int i = 0; i < splits.size(); i++) {
082        byteSplits[i] = Bytes.toBytes(splits.get(i));
083      }
084    }
085
086    log.info("Creating table");
087    try (Connection connection = ConnectionFactory.createConnection(HBaseConfiguration.create());
088         Admin admin = connection.getAdmin()) {
089
090      ColumnFamilyDescriptor cf = ColumnFamilyDescriptorBuilder
091        .newBuilder(colFamily.getBytes(StandardCharsets.UTF_8))
092        .setCompressionType(Compression.Algorithm.SNAPPY)
093        .setDataBlockEncoding(DataBlockEncoding.FAST_DIFF)
094        .setMaxVersions(1)
095        .build();
096      TableDescriptorBuilder ror = TableDescriptorBuilder
097        .newBuilder(TableName.valueOf(tableName))
098        .setColumnFamily(cf);
099
100      if (regionSize != null && regionSize > 0) {
101        ror.setMaxFileSize(regionSize * 1048576); // 1 MB = 2^20 bytes
102      }
103
104      if (byteSplits == null) {
105        admin.createTable(ror.build());
106      } else {
107        admin.createTable(ror.build(), byteSplits);
108      }
109    } catch (IOException e) {
110      log.info("HBase exception", e);
111      System.exit(1);
112    }
113    log.info("Table created - exiting");
114  }
115
116  public static void main(String[] args) {
117    if (args.length < 2) {
118      System.out.println("Usage: TableCreator <table_name> <column_family> [<splits_file>] [<region_file_size_mb>]");
119      System.exit(1);
120    }
121    String tableName = args[0];
122    String colFam = args[1];
123    String splitsFile = args.length > 2 ? args[2] : null;
124    Integer regionSize = args.length == 4 ? Integer.valueOf(args[3]) : null;
125
126    TableCreator instance = new TableCreator();
127    //    instance.createPreSplitTable("ror_120", "v", "/Users/oliver/SourceCode/ror_120_splits.txt", 256*3);
128    instance.createPreSplitTable(tableName, colFam, splitsFile, regionSize);
129  }
130}
