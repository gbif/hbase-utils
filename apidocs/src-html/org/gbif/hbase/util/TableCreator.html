001package org.gbif.hbase.util;
002
003import java.io.BufferedReader;
004import java.io.File;
005import java.io.FileInputStream;
006import java.io.FileNotFoundException;
007import java.io.IOException;
008import java.io.InputStreamReader;
009import java.io.UnsupportedEncodingException;
010import java.util.ArrayList;
011import java.util.List;
012
013import org.apache.hadoop.conf.Configuration;
014import org.apache.hadoop.hbase.HBaseConfiguration;
015import org.apache.hadoop.hbase.HColumnDescriptor;
016import org.apache.hadoop.hbase.HTableDescriptor;
017import org.apache.hadoop.hbase.MasterNotRunningException;
018import org.apache.hadoop.hbase.ZooKeeperConnectionException;
019import org.apache.hadoop.hbase.client.HBaseAdmin;
020import org.apache.hadoop.hbase.io.compress.Compression;
021import org.apache.hadoop.hbase.io.encoding.DataBlockEncoding;
022import org.apache.hadoop.hbase.util.Bytes;
023import org.slf4j.Logger;
024import org.slf4j.LoggerFactory;
025

027
028public class TableCreator {
029
030  private static Logger log = LoggerFactory.getLogger(org.gbif.hbase.util.TableCreator.class);
031
032  /**
033   * Create an hbase table with given colfamily, using the given splits file to presplit the table.  Regions are set
034   * to the given max regionSize (given in MB and translated into bytes). Input table
035   *
036   * @param tableName      table to create
037   * @param colFamily      col family to create
038   * @param splitsFileName one split per line, interpreted as ints
039   * @param regionSize     max region file size in MB
040   */
041  private void createPreSplitTable(String tableName, String colFamily, String splitsFileName, Integer regionSize) {
042    byte[][] byteSplits = null;
043
044    if (splitsFileName != null) {
045      // assumes splits are written as strings but are to be interpreted as ints
046      log.info("Reading splits file [{}]", splitsFileName);
047      List<Integer> splits = new ArrayList<Integer>();
048      try {
049        BufferedReader reader =
050          new BufferedReader(new InputStreamReader(new FileInputStream(new File(splitsFileName)), "UTF-8"));
051        while (reader.ready()) {
052          String line = reader.readLine();
053          splits.add(Integer.valueOf(line));
054        }
055      } catch (FileNotFoundException e) {
056        log.warn("Splits file not found - exiting", e);
057        System.exit(1);
058      } catch (UnsupportedEncodingException e) {
059        log.error("Drastic jvm failure - exiting", e);
060        System.exit(1);
061      } catch (IOException e) {
062        log.warn("Failure reading splits file - exiting", e);
063        System.exit(1);
064      }
065      log.info("Loaded [{}] splits", splits.size());
066
067      byteSplits = new byte[splits.size()][];
068      for (int i = 0; i < splits.size(); i++) {
069        byteSplits[i] = Bytes.toBytes(splits.get(i));
070      }
071    }
072
073    log.info("Creating table");
074    Configuration config = HBaseConfiguration.create();
075    try {
076      HBaseAdmin admin = new HBaseAdmin(config);
077      HColumnDescriptor cf = new HColumnDescriptor(colFamily);
078      cf.setCompressionType(Compression.Algorithm.SNAPPY);
079      cf.setDataBlockEncoding(DataBlockEncoding.FAST_DIFF);
080      cf.setMaxVersions(1);
081      HTableDescriptor ror = new HTableDescriptor(tableName);
082      if (regionSize != null && regionSize > 0) ror.setMaxFileSize(regionSize * 1048576); // 1 MB = 2^20 bytes
083      ror.addFamily(cf);
084      if (byteSplits == null) {
085        admin.createTable(ror);
086      }
087      else {
088        admin.createTable(ror, byteSplits);
089      }
090      admin.close();
091    } catch (MasterNotRunningException e) {
092      log.info("HBase exception", e);
093      System.exit(1);
094    } catch (ZooKeeperConnectionException e) {
095      log.info("HBase exception", e);
096      System.exit(1);
097    } catch (IOException e) {
098      log.info("HBase exception", e);
099      System.exit(1);
100    }
101    log.info("Table created - exiting");
102  }
103
104  public static void main(String[] args) {
105    if (args.length < 2) {
106      System.out.println("Usage: TableCreator <table_name> <column_family> [<splits_file>] [<region_file_size_mb>]");
107      System.exit(1);
108    }
109    String tableName = args[0];
110    String colFam = args[1];
111    String splitsFile = args.length > 2 ? args[2] : null;
112    Integer regionSize = args.length == 4 ? Integer.valueOf(args[3]) : null;
113
114    TableCreator instance = new TableCreator();
115    //    instance.createPreSplitTable("ror_120", "v", "/Users/oliver/SourceCode/ror_120_splits.txt", 256*3);
116    instance.createPreSplitTable(tableName, colFam, splitsFile, regionSize);
117  }
118}
